```dart
multiAngleCaptureController.capturedCount.addListener(() {
  if (multiAngleCaptureController.capturedCount.value == 3 && mounted) {
    // หยุด camera stream ก่อนเปิด bottom sheet
    cameraStreamController.stopImageStream();
    
    showModalBottomSheet(
      context: context,
      isScrollControlled: true,
      isDismissible: false, // ต้องกด Analyse หรือ Cancel
      builder: (_) => ArScanReviewBottomSheet(
        capturedImages: multiAngleCaptureController.capturedImages,
        // callback เมื่อกด Analyse → ส่งต่อให้ plan 10-02
        onAnalyse: (metadata) => _handleAnalyse(metadata),
        onCancel: () {
          // restart stream, reset controller
          multiAngleCaptureController.reset();
          cameraStreamController.startImageStream(onFrame);
          Navigator.pop(context);
        },
      ),
    );
  }
});
```
