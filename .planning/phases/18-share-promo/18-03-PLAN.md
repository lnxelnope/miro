```dart
class PromoCodeService {
  static const _baseUrl = 'https://us-central1-miro-d6856.cloudfunctions.net';

  /// Redeem a promo code. Returns parsed response map.
  /// Throws PromoRedeemException on failure with error key for L10n mapping.
  static Future<Map<String, dynamic>> redeemCode(String code) async {
    final deviceId = await DeviceIdService.getDeviceId();
    final response = await http.post(
      Uri.parse('$_baseUrl/redeemPromoCode'),
      headers: {'Content-Type': 'application/json'},
      body: jsonEncode({'deviceId': deviceId, 'code': code.trim()}),
    );

    final body = jsonDecode(response.body) as Map<String, dynamic>;

    if (response.statusCode == 200 && body['success'] == true) {
      return body;
    }

    // Map error key from API → throw with key for L10n
    throw PromoRedeemException(body['error'] as String? ?? 'unknown');
  }
}

class PromoRedeemException implements Exception {
  final String errorKey; // "invalid_code", "expired", "max_reached", "per_user_limit"
  PromoRedeemException(this.errorKey);
}
```
