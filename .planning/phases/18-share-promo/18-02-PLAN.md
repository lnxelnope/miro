สร้าง `functions/src/promo/redeemPromoCode.ts`:

```typescript
import { onRequest } from "firebase-functions/v2/https";
import * as admin from "firebase-admin";

if (!admin.apps.length) admin.initializeApp();
const db = admin.firestore();

export const redeemPromoCode = onRequest(
  { timeoutSeconds: 15, memory: "256MiB", cors: true },
  async (req, res) => {
    // 1. เช็ค POST
    if (req.method !== "POST") {
      res.status(405).json({ error: "Method not allowed" });
      return;
    }

    // 2. อ่าน body
    const { deviceId, code } = req.body;
    if (!deviceId || !code) {
      res.status(400).json({ error: "Missing deviceId or code" });
      return;
    }

    // 3. Normalize code
    const normalizedCode = code.trim().toUpperCase();

    // 4. ค้นหาโค้ดใน Firestore
    // query by code field (ไม่ใช่ doc ID)
    const snapshot = await db.collection("promo_codes")
      .where("code", "==", normalizedCode)
      .where("active", "==", true)
      .limit(1).get();

    if (snapshot.empty) {
      res.status(404).json({ error: "invalid_code", message: "Promo code not found or inactive" });
      return;
    }

    const promoDoc = snapshot.docs[0];
    const promo = promoDoc.data();

    // 5. เช็ค expiry
    if (promo.expiryAt && promo.expiryAt.toDate() < new Date()) {
      res.status(400).json({ error: "expired", message: "Promo code has expired" });
      return;
    }

    // 6. เช็ค maxRedemptions
    if (promo.maxRedemptions > 0 && promo.redemptionCount >= promo.maxRedemptions) {
      res.status(400).json({ error: "max_reached", message: "Promo code redemption limit reached" });
      return;
    }

    // 7. เช็ค perUserLimit (idempotent)
    const redemptionRef = promoDoc.ref.collection("redemptions").doc(deviceId);
    const redemptionSnap = await redemptionRef.get();
    if (redemptionSnap.exists) {
      const existing = redemptionSnap.data()!;
      if (existing.count >= promo.perUserLimit) {
        res.status(400).json({ error: "per_user_limit", message: "You have already redeemed this code" });
        return;
      }
    }

    // 8. Grant reward (Firestore transaction)
    // ตาม rewardType: energy → อัปเดต energy_balances, freepass → อัปเดต users
    // (ดูจาก claimFreepassEndpoint / syncBalance สำหรับ pattern)

    // 9. บันทึก redemption + increment count
    // 10. Response success
  }
);
```

**สำคัญ:**
- ใช้ Firestore **transaction** หรือ **batch write** เพื่อ atomic: increment `redemptionCount` + create/update redemption doc + grant reward
- reward logic ให้ดูจาก `claimFreepassEndpoint` (freepass) และ `syncBalance`/`addEnergy` (energy) เป็น pattern
- ไม่ log โค้ดเต็มใน console
- Response: `res.status(200).json({ success: true, rewardType, rewardPayload, message: "..." })`
