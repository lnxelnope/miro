String detectAssetType(String query) {
  final queryLower = query.toLowerCase().replaceAll('-', '').replaceAll('_', '');
  
  // Check Thai gold
  if (THAI_GOLD_KEYWORDS.contains(queryLower)) {
    return 'gold';

  for (var pattern in FUND_PATTERNS) {

    return 'stock';
  }
  
  return 'stock'; // default
