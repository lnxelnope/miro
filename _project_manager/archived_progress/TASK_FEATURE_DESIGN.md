  void scheduleNudges() {
    // Food logging nudges
    scheduleAt(time: "07:00", check: () => !hasLoggedBreakfast());

## 9. Data Models

### 9.1 Task

```dart
@collection
class Task {
  Id id = Isar.autoIncrement;
  
  late String title;
  String? description;
  
  @enumerated
  late TaskType type;
  
  @enumerated
  late TaskStatus status;
  
  @enumerated
  late TaskPriority priority;
  
  @enumerated
  late TaskSource source;
  
  // Timing (for calendar events)
  DateTime? dueDate;
  DateTime? dueTime;
  DateTime? endTime;
  bool isAllDay = false;
  
  // Checklist (for todo lists)
  List<ChecklistItem>? checklist;
  
  // Recurring
  RecurringPattern? recurring;
  
  // Linking
  String? category;             // health, finance, general
  int? linkedWorkoutDayId;      // Link to workout program
  int? linkedGoalId;            // Link to financial goal
  
  // Google Calendar
  String? googleEventId;
  String? googleCalendarLink;
  bool isSynced = false;
  
  // Completion
  DateTime? completedAt;
  String? completionNotes;
  
  // Meta
  List<String> tags = [];
  DateTime createdAt = DateTime.now();
  DateTime updatedAt = DateTime.now();
}

@embedded
class ChecklistItem {
  late String id;
  late String title;
  bool isCompleted = false;
}

@embedded
class RecurringPattern {
  late String frequency;        // daily, weekly, monthly
  late int interval;            // every X days/weeks/months
  List<int>? daysOfWeek;        // for weekly [0-6]
  int? dayOfMonth;              // for monthly
  DateTime? endDate;
}
```
